The following code is to answer research question 1 (Does tail filtering improve prediction accuracy and Sharpe ratios?) by comparing the average 1-month returns across three approaches: Long-Short Return (the proposed method stated in our paper), Equal-Weighted Market Return (serving as a benchmark), and Simple Tree Long-Short Return (using an unfiltered training dataset) based on quantile 0.15. The threshold q=0.15 is chosen based on the results of Research Question 2, where the filtered long-short portfolio achieved the highest average 1-month return when q=0.15. A decision tree model is then trained on the filtered dataset to predict 12-month forward returns (R12M_Usd). Model performance is evaluated by constructing equal-weighted long-short portfolios based on the predicted returns. The effectiveness of the model is assessed by calculating the average portfolio returns and conducting a t-test on the return spread between the top and bottom predicted categories. An equal-weighted market benchmark is also computed for comparison, enabling an assessment of whether tail filtering improves predictive accuracy and risk-adjusted returns.

library(xgboost)
library(dplyr)

# Setup
tick <- levels(data_ml$stock_id)        # IDs of stocks
dates <- levels(as.factor(data_ml$date))    # List of dates
oos <- "2007-12-06"
dates_oos <- dates[dates>oos]                       # Out-of-sample dates
Tt <- length(dates_oos)                             # Number of dates
nb_port <- 6                                        # Number of portfolios (hard-coded  be careful)
q_port <- 1 / nb_port                     
q_port <- seq(0, 1, by  = q_port)                   # Quantiles for portfolio construction
r_port <- matrix(0, ncol = nb_port, nrow = Tt)      # Portfolio returns
offset_date <- 0                                    # 0 => is equivalent to 9 years rolling windows
max_depth <- 5                                      # Maximum nb of levels for the trees
q <- 0.15                                           # Filtering quantile
ptm <- proc.time()                                  # Initiate CPU time

# Loop through each out-of-sample date
for (t in 1:Tt) {
  if (t %% 12 == 0) cat("Processing date:", dates_oos[t], "\n")
  
  # Rolling window filter
  x <- filter(data_ml,
              date < as.Date(dates_oos[t]) - 365,
              date > dates[t + offset_date])        #rolling window
  
  # Tail filter
  q_l <- quantile(x$R12M_Usd, q) 
  q_h <- quantile(x$R12M_Usd, 1 - q)
  x <- x[x$R12M_Usd <= q_l | x$R12M_Usd >= q_h, ]
  
  # Training
  train_data <- x[, !(names(x) %in% c("stock_id", "date", "R1M_Usd","R3M_Usd",
                                        "R6M_Usd", "R12M_Usd"))] %>% data.matrix() #exclude irrelevant columns  
  train_label <- x$R12M_Usd
  train_matrix <- xgb.DMatrix(data = train_data, label = train_label)
  
  fit <- xgb.train(
    data = train_matrix,
    objective = "reg:squarederror",
    eta = 0.7,                        # Learning rate, 0.7 for light, 0.5 for intermed. 0.3 for deep
    max_depth = max_depth,            # max_depth is set to 5
    gamma = 0,                        # penalization
    lambda = 1,                       # penalization
    nrounds = 25,                     # Nb of trees, 25 for light, 50 for intermed., 100 for deep
    nthread = parallel::detectCores()
  )
  
  # Prediction
  y <- filter(data_ml, date == dates_oos[t])
  test_data <- y[, !(names(y) %in% c("stock_id", "date", "R1M_Usd","R3M_Usd",
                                        "R6M_Usd", "R12M_Usd"))] %>% data.matrix()
  test_label <- y$R1M_Usd
  pred <- predict(fit, xgb.DMatrix(test_data))
  
  # Portfolio constructions
  pred_q <- quantile(pred, probs = q_port)
  for (j in 1:nb_port) {
    ind <- which(pred >= pred_q[j] & pred <= pred_q[j + 1])
    r_port[t, j] <- mean(test_label[ind], na.rm = TRUE)
  }
}

The following code is to answer Research Question 2 (What is the optimal quantile q in a given dataset?). We chose the quantiles to be 0.1, 0.15, 0.2, 0.25, and 0.3, which correspond to 20%, 30%, 40%, 50%, and 60% of the data, respectively. The code loops through each candidate quantile, applies a tail filter based on the selected q, trains a decision tree model to predict 12-month returns (R12M_Usd), and constructs an equal-weighted long-short portfolio based on the models predictions. For each quantile, the average 1-month long-short return (R1M_Usd) is computed to evaluate performance. The results are then compared across different quantiles to determine which q value leads to the best portfolio performance.

library(xgboost)
library(tidyverse)
# R1M_Usd
# Define the quantiles to test

q_candidates <- c(0.1, 0.15, 0.2, 0.25, 0.3)
results <- data.frame(Quantile = numeric(), AvgReturn = numeric())

# Save runtime
ptm <- proc.time()

# Loop through each quantile
for (q in q_candidates) {
  cat("Testing q =", q, "\n")
  
  r_port <- matrix(0, ncol = 6, nrow = length(dates[dates > oos]))
  
  for (t in 1:Tt) {
    if (t %% 12 == 0) cat("Processing date:", dates_oos[t], "\n")
  
    # Rolling window filter
    x <- filter(data_ml,
                date < as.Date(dates_oos[t]) - 365,
                date > dates[t + offset_date])
  
    
    # Filter to top and bottom q quantiles
    q_l <- quantile(x$R12M_Usd, q)
    q_h <- quantile(x$R12M_Usd, 1 - q)
    x <- x[x$R12M_Usd <= q_l | x$R12M_Usd >= q_h, ]
    
    train_data <- x[, !(names(x) %in% c("stock_id", "date", "R1M_Usd","R3M_Usd",
                                        "R6M_Usd", "R12M_Usd"))] %>% data.matrix()
    train_label <- x$R12M_Usd
    train_matrix <- xgb.DMatrix(data = train_data, label = train_label)
    
    fit <- xgb.train(
      data = train_matrix,
      objective = "reg:squarederror",
      eta = 0.7,
      max_depth = 5,
      gamma = 0,
      lambda = 1,
      nrounds = 25,
      nthread = parallel::detectCores()
    )
    
    y <- filter(data_ml, date == dates[dates > oos][t])
    test_data <- y[, !(names(y) %in% c("stock_id", "date", "R1M_Usd","R3M_Usd", 
                                       "R6M_Usd", "R12M_Usd"))] %>% data.matrix()
    test_label <- y$R1M_Usd
    pred <- predict(fit, xgb.DMatrix(data = test_data, label = test_label))
    
    pred_q <- quantile(pred, probs = seq(0, 1, length.out = 7))
    for (j in 1:6) {
      ind <- which(pred >= pred_q[j] & pred <= pred_q[j + 1])
      r_port[t, j] <- mean(test_label[ind], na.rm = TRUE)
    }
  }

  # Compute long-short return
  avg_return <- mean(r_port[,6] - r_port[,1], na.rm = TRUE)
  
  # Compute average training R12M return used in that quantile
  avg_R12M <- mean(x$R12M_Usd, na.rm = TRUE)

  # Store results
  results <- rbind(results, data.frame(
    Quantile = q,
    Avg_1M_LongShort = avg_return,
    Avg_12M_Train = avg_R12M
  ))
}

The following code is to answer Research Question 3 (Does tail filtering improve prediction accuracy or portfolio returns uniformly across different types of stocks, such as by firm size?). We aim to verify whether the size anomaly (the tendency for small firms to outperform large firms over very long periods) still holds when applying the tail-value method. The code applies a 15% tail filter on R12M_Usd to select extreme-performing stocks, trains a decision tree model on the filtered dataset, and predicts future returns. The stocks are then divided into small-cap and large-cap groups based on a median split of market capitalization. For each size group, equal-weighted long-short portfolios are constructed based on predicted returns, and the average long-short return and Sharpe ratio are computed. These results are used to assess whether the predictive performance of the tail filtering strategy differs between small and large firms.

library(xgboost)
library(tidyverse)

offset_date <- 0
# Set quantile
q <- 0.15

# Initialize containers
size_r_port <- list(SmallCap = data.frame(), LargeCap = data.frame())

# Loop through test dates
for (t in 1:length(dates[dates > oos])) {
  cat("Date:", dates[dates > oos][t], "\n")
  
  # Rolling training window 
  x <- filter(data_ml,
              date < as.Date(dates_oos[t]) - 365,
              date > dates[t + offset_date])
  
  # Filter to top and bottom q quantiles of R12M_Usd
  q_l <- quantile(x$R12M_Usd, q)
  q_h <- quantile(x$R12M_Usd, 1 - q)
  x <- x[x$R12M_Usd <= q_l | x$R12M_Usd >= q_h, ]
  
  train_data <- x[, !(names(x) %in% c("stock_id", "date", "R1M_Usd",
                                      "R3M_Usd", "R6M_Usd", "R12M_Usd"))] %>% 
                                      data.matrix()
  train_label <- x$R12M_Usd
  train_matrix <- xgb.DMatrix(data = train_data, label = train_label)
  
  # train descision tree
  fit <- xgb.train(
    data = train_matrix,
    objective = "reg:squarederror",
    eta = 0.7,
    max_depth = 5,
    gamma = 0,
    lambda = 1,
    nrounds = 25,
    nthread = parallel::detectCores()
  )
  
  # Test set for evaluation
  y <- filter(data_ml, date == dates[dates > oos][t])
  test_data <- y[, !(names(y) %in% c("stock_id", "date", "R1M_Usd", 
                                     "R3M_Usd", "R6M_Usd", "R12M_Usd"))] %>% data.matrix()
  pred <- predict(fit, xgb.DMatrix(test_data))
  
  y$pred <- pred
  
  #Size Grouping: SmallCap vs LargeCap (by median split of market cap)

  y$size_group <- ifelse(y$Mkt_Cap_12M_Usd <= median(y$Mkt_Cap_12M_Usd, na.rm = TRUE), 
                         "SmallCap", "LargeCap")
  
  for (g in c("SmallCap", "LargeCap")) {
    y_group <- y[y$size_group == g, ]
    
    # Quantile predictions within size group
    pred_q <- quantile(y_group$pred, probs = seq(0, 1, length.out = 7), na.rm = TRUE)
    
    port_ret <- sapply(1:6, function(j) {
      ind <- which(y_group$pred >= pred_q[j] & y_group$pred <= pred_q[j + 1])
      mean(y_group$R1M_Usd[ind], na.rm = TRUE)
    })
    
    size_r_port[[g]] <- rbind(size_r_port[[g]], data.frame(
      date = as.character(dates[dates > oos][t]),
      group = g,
      P1 = port_ret[1],
      P6 = port_ret[6],
      LongShort = port_ret[6] - port_ret[1]
    ))
  }
}
